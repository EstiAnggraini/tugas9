extension DateTimeExtensions on DateTime {
  // Method
  String toFormattedString() {
    return '$day/$month/$year';
  }
  
  // Getter
  bool get isToday {
    var now = DateTime.now();
    return year == now.year && 
           month == now.month && 
           day == now.day;
  }
  
  // Operator
  DateTime operator +(Duration duration) {
    return add(duration);
  }
}

void main() {
  // 1. Implementasi toFormattedString()
  print('--- toFormattedString() ---');
  final now = DateTime.now();
  print('Current date: ${now.toFormattedString()}');

  final specificDate = DateTime(2023, 10, 26);
  print('Specific date (2023-10-26): ${specificDate.toFormattedString()}');

  print('\n--- isToday ---');
  // 2. Implementasi isToday
  final today = DateTime.now();
  final tomorrow = DateTime.now().add(Duration(days: 1));
  final yesterday = DateTime.now().subtract(Duration(days: 1));

  print('Is ${today.toFormattedString()} today? ${today.isToday}');
  print('Is ${tomorrow.toFormattedString()} today? ${tomorrow.isToday}');
  print('Is ${yesterday.toFormattedString()} today? ${yesterday.isToday}');

  print('\n--- operator + ---');
  // 3. Implementasi operator +
  final initialDate = DateTime(2024, 1, 15);
  final durationToAdd = Duration(days: 10, hours: 3);
  final futureDate = initialDate + durationToAdd;

  print('Initial date: ${initialDate.toFormattedString()}');
  print('Duration to add: ${durationToAdd.inDays} days, ${durationToAdd.inHours % 24} hours');
  print('Future date: ${futureDate.toFormattedString()}');
  print('Future date (full): $futureDate');

  final anotherDuration = Duration(days: 2);
  final evenFutureDate = futureDate + anotherDuration;
  print('Future date + 2 day: ${evenFutureDate.toFormattedString()}');
}

extension NullableStringExtensions on String? {
// Safe access untuk nullable string
String orEmpty() => this ?? '';
bool get isNullOrEmpty => this == null || this!.isEmpty;
bool get isNotNullOrEmpty => !isNullOrEmpty;
String withDefault(String defaultValue) => this ?? defaultValue;
}
void main(){

print(name.isNotNullOrEmpty); // false
print(name.withDefault('Guest')); // 'Guest'
}

extension NumberExtensions on num {
  // Format ke Rupiah
  String toRupiah() {
    return 'Rp${toStringAsFixed(0).replaceAllMapped(
      RegExp(r'(\d{1,3})(?=(\d{3})+(?!\d))'),
      (match) => '${match[1]}.',
    )}';
  }
  
  // Check jika dalam range
  bool isBetween(num min, num max) {
    return this >= min && this <= max;
  }
  
  // Pangkat
  num power(int exponent) {
    var result = this;
    for (var i = 1; i < exponent; i++) {
      result *= this;
    }
    return result;
  }
}

void main(){
// Penggunaan
var price = 1500000;
print(price.toRupiah());  // 'Rp1.500.000'
print(5.isBetween(1, 10)); // true
print(2.power(3));         // 8
}

extension ListExtensions on List {
  // Safe get dengan default
  T? getOrNull(int index) {
    if (index < 0 || index >= length) return null;
    return this[index];
  }
  
  // Split list menjadi chunks
  List> chunk(int size) {
    var chunks = >[];
    for (var i = 0; i < length; i += size) {
      chunks.add(sublist(i, i + size > length ? length : i + size));
    }
    return chunks;
  }
  
  // Distinct items
  List distinct() {
    var result = [];
    for (var item in this) {
      if (!result.contains(item)) {
        result.add(item);
      }
    }
    return result;
  }
}

void main(){
// Penggunaan
var numbers = [1, 2, 3, 4, 5, 6, 7];
print(numbers.chunk(3));  // [[1, 2, 3], [4, 5, 6], [7]]
print([1, 2, 2, 3, 3, 3].distinct()); // [1, 2, 3]
}

  class Product {
  final String name;
  final double price;
  
  Product(this.name, this.price);
}

// Extension untuk formatting double ke Rupiah
extension DoubleRupiahFormatter on double {
  String toRupiah() {
    // Convert to integer to handle whole Rupiah amounts without decimals
    final intValue = toInt();
    
    // Convert to string
    String numberString = intValue.toString();
    
    // Add thousands separators
    String formattedString = '';
    int counter = 0;
    for (int i = numberString.length - 1; i >= 0; i--) {
      formattedString = numberString[i] + formattedString;
      counter++;
      if (counter % 3 == 0 && i != 0) {
        formattedString = '.' + formattedString;
      }
    }
    
    return 'Rp$formattedString';
  }
}

// Extension untuk Product
extension ProductExtensions on Product {
  // Method untuk diskon
  Product withDiscount(double percentage) {
    var discountedPrice = price * (1 - percentage / 100);
    return Product(name, discountedPrice);
  }
  
  // Format untuk display
  String get displayInfo => '$name - ${price.toRupiah()}';
  
  // Check jika mahal
  bool get isExpensive => price > 1000000;
}

void main(){
  // Penggunaan
  var laptop = Product('Laptop', 15000000);
  var discounted = laptop.withDiscount(10);
  print(discounted.displayInfo);  // 'Laptop - Rp13.500.000'
  print(laptop.isExpensive);      // true

  var keyboard = Product('Keyboard', 750000);
  print(keyboard.displayInfo);
  print(keyboard.isExpensive);
}

extension StringValidators on String {
  bool get isEmail {
    return RegExp(r'^[\w-\.]+@([\w-]+\.)+[\w-]{2,4}$').hasMatch(this);
  }

  bool get isPhoneNumber {
    return RegExp(r'^[0-9]{10,13}$').hasMatch(this);
  }

  bool get isStrongPassword {
    return length >= 8 &&
        contains(RegExp(r'[A-Z]')) &&
        contains(RegExp(r'[a-z]')) &&
        contains(RegExp(r'[0-9]'));
  }

  bool get isUrl {
    try {
      Uri.parse(this);
      return true;
    } catch (_) {
      return false;
    }
  }
}

extension IntValidators on int {
  bool get isAdultAge => this >= 18;
  bool get isPositive => this > 0;
  bool get isEven => this % 2 == 0;
  bool get isOdd => this % 2 != 0;
}

void main() {
  // Penggunaan
  var email = 'test@example.com';
  var age = 25;
  var password = 'Pass1234';

  print(email.isEmail); // true
  print(age.isAdultAge); // true
  print(password.isStrongPassword); // true
  print(10.isEven); // true
}

🏝️ Ringkasan & Best Practices

🌿 Ringkasan:

  • Extension methods: tambah method ke tipe yang ada
  • Sintaks: extension Nama on Type
  • Support: method, getter, setter, operator
  • Generic extensions: on List<T>

✅ Best Practices:

  • 📛 Nama jelas: StringExtensions, DateTimeUtils
  • 📁 Group by type: Satu file per tipe utama
  • 🚫 Avoid overuse: Jangan jadikan semua utility sebagai extension
  • 🛡️ Null safety: Handle nullable types dengan baik
  • 📝 Dokumentasi: Tambahkan komentar untuk public extensions
🌱 "Jaga kode tetap alami, berdayakan ekstensi tanpa merusak aslinya — seperti hutan yang tumbuh harmonis."

Saat kode dijalankan, akan terjadi ERROR kompilasi

📖 Penjelasan Penyebab:
MasalahKedua extension memiliki method dengan nama yang sama yaitu toInt()
Return type berbedaSatu return int, satu return double
Dart tidak bisa membedakanCompiler bingung method toInt() mana yang dipanggil

tag tag-success">✅ SOLUSI: Ubah Nama Method (Pilihan)

// Nama: Esti Anggraini
// Nim: 251410031
// Kelas: SI2A                 
extension StringNumber on String {
  int toInt() {
    return int.parse(this);
  }
}

extension StringDouble on String {
  double toDouble() {  // ✅ Ganti nama jadi toDouble()
    return double.parse(this);
  }
}

void main() {
  String angka = '123';
  print(angka.toInt());     // Output: 123
  print(angka.toDouble());  // Output: 123.0
}

✅ Kesimpulan Soal 1: Solusi terbaik adalah mengganti nama method menjadi toDouble() agar tidak terjadi konflik dengan method toInt().

  }

  c.increment();

⚠️ Apa yang Terjadi?

Saat kode dikompilasi, akan terjadi ERROR kompilasi

📖 Penjelasan Penyebab:
PenyebabPenjelasan
Private Member Di Dart, identifier yang diawali underscore _value bersifat private (hanya bisa diakses di library yang sama)
Extension Berbeda Library Extension CounterStats berada di luar library class Counter, sehingga tidak memiliki akses ke member private
Prinsip Encapsulation Private member adalah implementasi detail yang tidak boleh diakses dari luar class
Extension Bukan Bagian Class Extension hanyalah "tambahan" dari luar, bukan bagian internal class, sehingga tidak punya hak istimewa

tag tag-success">✅ SOLUSI: Tambahkan Getter Publik (Rekomendasi)

// Nama: Esti Anggraini
// Nim: 251410031
// Kelas: SI2A                 
class Counter {
  int _value = 0;  // ✅ TETAP PRIVATE (tidak diubah jadi publik)

  void increment() {
    _value++;
  }
  
  // ✅ Tambahkan getter publik (tanpa mengubah _value jadi publik)
  int get value => _value;
}

extension CounterStats on Counter {
  int getValue() {
    return value;  // ✅ Akses melalui getter publik
  }
}

void main() {
  var c = Counter();
  c.increment();
  c.increment();
  c.increment();
  print('Nilai counter: ${c.getValue()}'); // Output: 3
}
